#import clean
data("ny_noaa")

noaa = ny_noaa %>%
  janitor::clean_names()

noaa = 
  noaa %>% 
  separate(date, into = c("year", "month", "day"), convert = TRUE) %>% 
  select(
    -id, -day, -tmax
  ) %>%
  filter(
    year >=2000 & year <= 2010
  ) %>% #keep interested time interval
  drop_na(c(snow,prcp,snwd,tmin))
#convert units
noaa = 
  noaa %>%  
  mutate(
      tmin = as.numeric(tmin),
      snow_mm = snow,
      snow = round(snow_mm * 0.03937 * 4) / 4) %>%
  select(
    -snow
  )

Scatter Plot to Show the General Snowfall from 2000 to 2010

noaa %>%
  group_by(year,month)%>%
  summarize(Snowfall = mean(snow_mm))%>%
  plot_ly(
      x = ~year, y = ~Snowfall, type = "scatter", color = ~month, alpha = 0.5)
